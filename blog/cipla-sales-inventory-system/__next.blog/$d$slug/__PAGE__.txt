class LoanScoringEngine:
    def __init__(self, criteria_config):
        self.criteria = criteria_config
        self.models = self.load_ensemble_models()

    def evaluate_applicant(self, applicant_data):
        # Traditional credit score (if available)
        credit_score = self.get_bureau_score(applicant_data)

        # Alternative data scoring for thin-file customers
        if credit_score is None:
            alt_score = self.calculate_alternative_score(
                utility_payments=applicant_data['utilities'],
                income_proxies=applicant_data['income_signals'],
                behavioral_data=applicant_data['transactions']
            )
            return self.apply_criteria(alt_score, 'alternative')

        return self.apply_criteria(credit_score, 'traditional')
